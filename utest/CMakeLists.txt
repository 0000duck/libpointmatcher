enable_testing()
find_package(GTest)
if(GTEST_FOUND)
	include_directories(${GTEST_INCLUDE_DIR})
	add_executable(utest utest.cpp)
	target_link_libraries(utest pointmatcher ${NABO_LIBRARY} ${EXTRA_LIBS} ${GTEST_LIBRARY} ${Boost_LIBRARIES} ${GTEST_LIBRARIES})
	add_test(utest ${CMAKE_CURRENT_BINARY_DIR}/utest --path "${CMAKE_SOURCE_DIR}/examples/data/")
else(GTEST_FOUND)
	message("Google test (gtest) not found - Skipping unit tests")
endif(GTEST_FOUND)
